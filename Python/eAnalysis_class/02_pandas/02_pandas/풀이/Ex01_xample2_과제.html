# big2 political party
big2 = df[(df['Political Party'] == 'Republican') | (df['Political Party'] == 'Democrat')]

numeric_columns = ['Years in office', 'Age at inauguration', 'Rating points', '% electoral', '% popular']

plt.figure(figsize=(16, 15)); gridsize=((len(numeric_columns) + 1) // 2, 2)
k = 0
for i in range(0, gridsize[0]):
    for j in range(0, gridsize[1]):    
        globals()['ax{}'.format(k+1)] = plt.subplot2grid(gridsize, (i, j))
        k += 1

for i, column in enumerate(numeric_columns):
    big2[[column, 'Political Party']].groupby(['Political Party']).mean().plot(kind='bar', rot=0, ax=globals()['ax{}'.format(i+1)])
plt.show()
